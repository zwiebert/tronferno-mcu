131  if (*dst && std::strcmp(*dst, src) == 0)
132  return true;

134  size_t src_size = std::strlen(src) + 1;

143  *dst = std::strcpy(ptr, src);
144  return true;

146  } else {
147  csu_destroy(dst);

152  }
