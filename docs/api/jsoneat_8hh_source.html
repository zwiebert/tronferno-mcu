383  /**

385  *
386  * \param buf destination buffer to copy to

389  * \return pointer to buffer on success. null-pointer on failure

391  char* copy_string(char *buf, size_t buf_size, pointer ptr) const {
392  size_t str_length = ptr->end - ptr->start;
