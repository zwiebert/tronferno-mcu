203  * \brief Build data and, while building, write them in chunks using a user provided callback function object (for http req, web-socket, ...)
204  *
205  * The function object may be created using std::bind to hold any data needed (e.g. file handles/descriptors, http requests).
206  */
207 class UoutWriterFunction final: public UoutWriter {
208 public:
209  /**
210  * \brief callback function object
211  * \param src Data to be written
212  * \param src_len Data length or -1 for zero terminated strings
213  * \param chunk_status Chunk number of data, to control output (e.g. 4 chunk sequence: -1,-2,-3,+4)
214  * - 0 for a single (first and final) "chunk"
215  * - -1 is the first chunk
216  * - -2 ... -n are in-between chunks
217  * - +n is the final chunk
218  * \return number of bytes written. If zero, the builder will just grow its buffer and may call again later.
219  */
220  using write_callback_type = std::function<int(const char *src,
221  ssize_t src_len, int chunk_status)>;
222 public:
223 
224  /**
225  * \param write_callback Callback function object for writing out our data
226  * \param fmt specify which data formats should be build by mySo()
227  */
228  UoutWriterFunction(write_callback_type write_callback, so_target_bits fmt = SO_TGT_FLAG_JSON) :
229  UoutWriter(fmt), m_write_callback(write_callback) {
230  }
231 
232  UoutWriterFunction(const UoutWriterFunction&) = delete;
233  virtual ~UoutWriterFunction() {
234  mySj.write_json(true); // write all remaining json
235  }
236 
237 private:
238  virtual int priv_write(const char *src, ssize_t src_len, bool is_final) override {
239  bool no_fragments = is_final && !nmb_chunks_written;
240 
241  int chunk_status = no_fragments ? 0 : (is_final ? (nmb_chunks_written + 1) : (nmb_chunks_written + 1) * -1);
242 
243  int n = m_write_callback(src, src_len, chunk_status);
244 
245  if (n > 0)
246  ++nmb_chunks_written;
247 
248  if (is_final)
249  nmb_chunks_written = 0;
250 
251  return n;
252  }
253 
254 private:
255  write_callback_type m_write_callback;
256  int nmb_chunks_written = 0;
257 };
