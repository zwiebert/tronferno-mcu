45  /// \return false if end was reached
46  operator bool() const {
47  return 0 == (mGm & ~0x3f);
48  }
49  /// \brief Increment counter
50  Fer_Gm_Counter& operator++() {
51  if ((mGm & ~0x3f) == 0) {
52  ++mGm;

55  ++mGm;

57  return *this;

66  return mGm >> 3;
67  }
